🔧 CUSTOMIZATION
═══════════════════════════════════════════════════════════════════════

All customizable via database:

Bot Name:
  UPDATE bot_config SET config_value = 'YourBot' 
  WHERE config_key = 'bot_name';

Welcome Message:
  UPDATE bot_config SET config_value = 'Hello!' 
  WHERE config_key = 'welcome_message';

Add FAQ:
  INSERT INTO knowledge_base (category, question, answer, keywords)
  VALUES ('company', 'Question?', 'Answer', '["keyword1", "keyword2"]');

Add Intent:
  INSERT INTO bot_intents (intent_name, display_name, training_phrases)
  VALUES ('custom', 'Custom Intent', '["phrase1", "phrase2"]');
